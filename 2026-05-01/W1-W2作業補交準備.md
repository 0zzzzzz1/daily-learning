## 作業 1｜W1-4/23：Vue 3 任務清單元件（`TaskList.vue`）

**作業規格回顧**：
- 使用 Vue 3 Composition API + TypeScript + Tailwind CSS
- `reactive` 管理任務列表，`ref` 管理輸入框
- `computed` 計算統計資訊（共 N 筆 / 已完成 M 筆）
- `watch` 監聽篩選條件，輸出切換 log

**參考實作**：

```vue
<!-- 2026-04-23/TaskList.vue -->
<script setup lang="ts">
import { ref, reactive, computed, watch } from 'vue'

// ── 型別定義 ─────────────────────────────────────────────
interface Task {
  id: number
  name: string
  completed: boolean
  createdAt: Date
}

type FilterType = 'all' | 'active' | 'completed'

// ── 響應式資料 ────────────────────────────────────────────
const newTaskName = ref<string>('')       // ref：單一基本型別
const filter = ref<FilterType>('all')
const errorMsg = ref<string>('')

const state = reactive<{                  // reactive：物件結構
  tasks: Task[]
  nextId: number
}>({
  tasks: [],
  nextId: 1,
})

// ── Computed：統計資訊 ─────────────────────────────────────
const totalCount = computed<number>(() => state.tasks.length)
const completedCount = computed<number>(
  () => state.tasks.filter(t => t.completed).length
)

const filteredTasks = computed<Task[]>(() => {
  switch (filter.value) {
    case 'active':    return state.tasks.filter(t => !t.completed)
    case 'completed': return state.tasks.filter(t => t.completed)
    default:          return state.tasks
  }
})

// ── Watch：監聽篩選條件切換 ────────────────────────────────
watch(filter, (newFilter, oldFilter) => {
  console.log(`[TaskList] 篩選條件切換：${oldFilter} → ${newFilter}`)
})

// ── 操作函式 ──────────────────────────────────────────────
const addTask = (): void => {
  const name = newTaskName.value.trim()
  if (!name) {
    errorMsg.value = '任務名稱不可為空'
    return
  }
  errorMsg.value = ''
  state.tasks.push({
    id: state.nextId++,
    name,
    completed: false,
    createdAt: new Date(),
  })
  newTaskName.value = ''
}

const toggleTask = (id: number): void => {
  const task = state.tasks.find(t => t.id === id)
  if (task) task.completed = !task.completed
}

const deleteTask = (id: number): void => {
  const index = state.tasks.findIndex(t => t.id === id)
  if (index !== -1) state.tasks.splice(index, 1)
}
</script>

<template>
  <div class="max-w-lg mx-auto p-6">
    <!-- 標題列 -->
    <div class="flex items-baseline gap-3 mb-4">
      <h1 class="text-2xl font-bold text-gray-800">📋 任務清單</h1>
      <span class="text-sm text-gray-500">
        共 {{ totalCount }} 筆 / 已完成 {{ completedCount }} 筆
      </span>
    </div>

    <!-- 輸入框 -->
    <div class="flex gap-2 mb-1">
      <input
        v-model="newTaskName"
        type="text"
        placeholder="輸入任務名稱..."
        class="flex-1 border border-gray-300 rounded-lg px-3 py-2 text-sm
               focus:outline-none focus:ring-2 focus:ring-indigo-400"
        @keydown.enter="addTask"
      />
      <button
        class="px-4 py-2 bg-indigo-600 text-white rounded-lg text-sm
               hover:bg-indigo-700 transition-colors"
        @click="addTask"
      >
        新增
      </button>
    </div>
    <p v-if="errorMsg" class="text-red-500 text-xs mb-2 ml-1">{{ errorMsg }}</p>

    <!-- 篩選按鈕 -->
    <div class="flex gap-2 mb-4 mt-3">
      <button
        v-for="(label, key) in ({ all: '全部', active: '未完成', completed: '已完成' } as Record<FilterType, string>)"
        :key="key"
        class="px-3 py-1 rounded-full text-xs border transition-colors"
        :class="filter === key
          ? 'bg-indigo-600 border-indigo-600 text-white'
          : 'border-gray-300 text-gray-600 hover:border-indigo-400 hover:text-indigo-600'"
        @click="filter = key"
      >
        {{ label }}
      </button>
    </div>

    <!-- 任務列表 -->
    <ul class="space-y-2">
      <li
        v-for="task in filteredTasks"
        :key="task.id"
        class="flex items-center gap-3 p-3 bg-white border border-gray-200
               rounded-lg hover:border-indigo-300 transition-colors group"
      >
        <input
          type="checkbox"
          :checked="task.completed"
          class="w-4 h-4 accent-indigo-600 cursor-pointer flex-shrink-0"
          @change="toggleTask(task.id)"
        />
        <span
          class="flex-1 text-sm"
          :class="task.completed ? 'line-through text-gray-400' : 'text-gray-800'"
        >
          {{ task.name }}
        </span>
        <button
          class="text-gray-300 hover:text-red-500 text-xl leading-none
                 transition-colors opacity-0 group-hover:opacity-100"
          @click="deleteTask(task.id)"
          title="刪除任務"
        >
          ×
        </button>
      </li>

      <!-- 空狀態 -->
      <li
        v-if="filteredTasks.length === 0"
        class="text-center text-gray-400 text-sm py-8"
      >
        {{ filter === 'all' ? '還沒有任何任務，新增一個吧！' : '目前沒有符合條件的任務' }}
      </li>
    </ul>
  </div>
</template>
```

// 2026-04-24/user-utils.ts

// ── 型別定義 ─────────────────────────────────────────────
interface RawUser {
  id: number
  name: string
  email: string
  password: string
  role: 'admin' | 'user'
  profile: {
    age: number
    city: string
    avatar?: string
  }
  permissions: string[]
}

// Omit<T, K>：工具型別，排除指定欄位
type SafeUser = Omit<RawUser, 'password'>

// ── 1. sanitizeUser：移除敏感欄位 ─────────────────────────
// 使用 Rest 語法 `{ password, ...rest }` 解構出其餘欄位
export const sanitizeUser = ({ password: _omitted, ...safeUser }: RawUser): SafeUser => {
  return safeUser
}
// 注意：`_omitted` 前綴底線表示「刻意忽略此變數」，避免 TypeScript unused variable 警告

// ── 2. extractProfile：巢狀解構取出常用欄位 ───────────────
export const extractProfile = ({
  name: userName,
  profile: { age: userAge, city: userCity },
}: RawUser): { userName: string; userAge: number; userCity: string } => {
  return { userName, userAge, userCity }
}

// ── 3. mergeUsers：淺合併 + 巢狀 profile 合併 ───────────
// 關鍵：`profile` 需額外合併，否則 Spread 會整個覆蓋
export const mergeUsers = (base: RawUser, patch: Partial<RawUser>): RawUser => {
  return {
    ...base,
    ...patch,
    profile: patch.profile
      ? { ...base.profile, ...patch.profile }   // 巢狀合併
      : base.profile,
  }
}
// 常見陷阱：`{ ...base, ...patch }` 會讓 patch.profile 整個覆蓋 base.profile，
// 導致 base.profile.age 等欄位遺失。必須顯式處理巢狀物件。

// ── 4. combinePermissions：合併多組權限（去重）────────────
// Rest 參數（`...groups`）接收任意數量的陣列參數
export const combinePermissions = (...groups: string[][]): string[] => {
  return [...new Set(groups.flat())]
}
// `groups.flat()` 展平二維陣列 → `new Set(...)` 去除重複 → `[...Set]` 轉回陣列

// ── 5. cloneUserWithDefaults：帶預設值的使用者複製 ─────────
export const cloneUserWithDefaults = (
  user: Partial<RawUser>,
  defaults: Partial<RawUser> = {}
): Partial<RawUser> => {
  return { ...defaults, ...user }  // user 的值優先覆蓋 defaults
}

// ── 測試案例（用於驗證實作正確性）────────────────────────
const mockUser: RawUser = {
  id: 1,
  name: 'Alice',
  email: 'alice@example.com',
  password: 'secret123',
  role: 'admin',
  profile: { age: 25, city: 'Taipei', avatar: 'https://example.com/avatar.png' },
  permissions: ['read', 'write'],
}

// 驗證 1：sanitizeUser 不含 password
const safe = sanitizeUser(mockUser)
console.assert(!('password' in safe), '❌ password 應被移除')
console.log('✅ sanitizeUser:', safe)

// 驗證 2：extractProfile 巢狀解構
const profile = extractProfile(mockUser)
console.assert(profile.userName === 'Alice', '❌ userName 應為 Alice')
console.assert(profile.userAge === 25, '❌ userAge 應為 25')
console.log('✅ extractProfile:', profile)

// 驗證 3：mergeUsers 巢狀 profile 合併
const merged = mergeUsers(mockUser, {
  name: 'Bob',
  profile: { age: 30, city: 'Tokyo' }
})
console.assert(merged.name === 'Bob', '❌ name 應為 Bob')
console.assert(merged.profile.age === 30, '❌ profile.age 應更新')
console.assert(merged.profile.avatar !== undefined, '❌ avatar 不應遺失')
console.log('✅ mergeUsers:', merged)

// 驗證 4：combinePermissions 去重
const combined = combinePermissions(['read', 'write'], ['write', 'delete'], ['read'])
console.assert(combined.length === 3, '❌ 應去除重複，共 3 個')
console.log('✅ combinePermissions:', combined)
```

// 2026-04-27/property-formatter.ts

// ── 型別定義 ─────────────────────────────────────────────
interface Property {
  id: number
  title: string
  address: {
    city: string
    district: string
    street: string
  }
  price: number
  area: number
  type: 'apartment' | 'house' | 'studio'
  features: string[]
  isAvailable: boolean
  listedAt: Date
}

const TYPE_LABELS: Record<Property['type'], string> = {
  apartment: '公寓',
  house:     '透天',
  studio:    '套房',
}

// ── 1. formatAddress：格式化完整地址 ─────────────────────
// 箭頭函式 + 參數解構 + Template Literal
export const formatAddress = ({
  address: { city, district, street }
}: Property): string =>
  `${city}${district}${street}`

// ── 2. formatPrice：格式化房價 ───────────────────────────
// 箭頭函式 + 預設參數 + 條件 Template Literal
export const formatPrice = (price: number, unit: string = '萬'): string => {
  if (unit === '萬' && price >= 10000) {
    const yi = price / 10000
    // 整除顯示整數，否則顯示一位小數
    return `NT$ ${yi % 1 === 0 ? yi : yi.toFixed(1)} 億`
  }
  return `NT$ ${price.toLocaleString('zh-TW')} ${unit}`
}

// ── 3. createPropertySummary：產生物件摘要 ───────────────
// 多行 Template Literal + 組合 formatAddress 和 formatPrice
export const createPropertySummary = (property: Property): string => {
  const { title: _title, area, type, features, isAvailable, price } = property
  const unitPrice = (price / area).toFixed(2)
  const featureList = features.join('、')
  const statusText = isAvailable ? '供應中' : '已售出'

  return `【${TYPE_LABELS[type]}】${formatAddress(property)}
售價：${formatPrice(price)} | 坪數：${area} 坪 | 單價：NT$ ${unitPrice} 萬/坪
特色：${featureList}
狀態：${statusText}`
}

// ── 4. createBatchReport：批次產生報表 ──────────────────
export const createBatchReport = (properties: Property[]): string => {
  const availableCount = properties.filter(p => p.isAvailable).length
  const avgPrice = properties.reduce((sum, p) => sum + p.price, 0) / properties.length

  const lines = properties.map((p, i) =>
    `${i + 1}. ${p.title}（${formatAddress(p)}）\n   ` +
    `${formatPrice(p.price)} | ${p.area} 坪 | ` +
    `${p.isAvailable ? '✅ 供應中' : '❌ 已售出'}`
  )

  return `=== 房源批次報告 ===
總筆數：${properties.length} 筆 | 可售：${availableCount} 筆 | 平均售價：${formatPrice(Math.round(avgPrice))}

${lines.join('\n')}`
}

// ── 測試案例 ─────────────────────────────────────────────
const sampleProperty: Property = {
  id: 1,
  title: '信義區精裝三房',
  address: { city: '台北市', district: '信義區', street: '忠孝東路一段10號' },
  price: 2500,
  area: 35,
  type: 'apartment',
  features: ['電梯', '車位', '近捷運'],
  isAvailable: true,
  listedAt: new Date('2026-04-01'),
}

console.log('formatAddress:', formatAddress(sampleProperty))
// 台北市信義區忠孝東路一段10號

console.log('formatPrice(2500):', formatPrice(2500))
// NT$ 2,500 萬

console.log('formatPrice(25000):', formatPrice(25000))
// NT$ 2.5 億

console.log('createPropertySummary:')
console.log(createPropertySummary(sampleProperty))
// 【公寓】台北市信義區忠孝東路一段10號
// 售價：NT$ 2,500 萬 | 坪數：35 坪 | 單價：NT$ 71.43 萬/坪
// 特色：電梯、車位、近捷運
// 狀態：供應中
```

## 本日技術複習

### 三大作業的核心語法對照

| 語法 | 作業 1 | 作業 2 | 作業 3 |
|------|--------|--------|--------|
| `ref` / `reactive` | ✅ 核心 | — | — |
| 解構賦值 | Props 解構 | ✅ 核心 | 參數解構 |
| Spread / Rest | `...state.tasks` | ✅ 核心 | — |
| 箭頭函式 | 所有 method | 所有函式 | ✅ 核心 |
| Template Literal | — | — | ✅ 核心 |
| `computed` | ✅ 核心 | — | — |
| `watch` | ✅ 核心 | — | — |
| TypeScript 型別 | interface | Omit<T,K> | Record<K,V> |

### 常見失誤提醒

1. **`reactive` 解構後失去響應性** → 須用 `toRefs()` 或改用 `ref`
2. **`mergeUsers` 的巢狀物件合併** → `{ ...base, ...patch }` 不夠，`profile` 需額外展開
3. **箭頭函式回傳物件字面量** → 需用括號包裹 `() => ({ key: value })`
4. **Template Literal 中的計算** → 可直接嵌入運算式 `${(a / b).toFixed(2)}`
